在调用 Async 的 read_with 的时候，先判断返回是不是 WouldBlockpub async fn read_with<R>(&self, op: impl FnMut(&T) -> io::Result<R>) -> io::Result<R> {
  let mut op = op;
  loop {
    match op(self.get_ref()) {
      Err(err) if err.kind() == io::ErrorKind::WouldBlock => {}
      res => return res,
    }
    optimistic(self.readable()).await?;
  }
}
